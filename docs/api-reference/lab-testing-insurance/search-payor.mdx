```java Java
import com.vital.api.Vital;
import com.vital.api.core.Environment;
import com.vital.api.resources.insurance.requests.PayorSearchRequest;

Vital vital = Vital.builder()
        .apiKey("YOUR_API_KEY")
        .environment(Environment.SANDBOX)
        .build();

PayorSearchRequest request = PayorSearchRequest
        .builder()
        .insuranceName("<insurance_name>")
        .build();

var data = vital.insurance().searchPayorInfo(request);
```

```go Go
import (
  "context"

  vital "github.com/tryVital/vital-go"
  vitalclient "github.com/tryVital/vital-go/client"
)

client := vitalclient.NewClient(
  vitalclient.WithApiKey("<YOUR_API_KEY>"),
  vitalclient.WithBaseURL(vital.Environments.Sandbox),
)

request := &vital.PayorSearchRequest{
    InsuranceName: "<insurance_name>",
}

response, err := client.Insurance.SearchPayorInfo(context.TODO(), request)
if err != nil {
    return err
}
fmt.Printf("Received data %s\n", response)
```
