  vitalclient.WithApiKey("<YOUR_API_KEY>"),
  vitalclient.WithBaseURL(vital.Environments.Sandbox),

    State:     "NY",
    ZipCode:   "10001",

response, err := client.LabTests.RequestPhlebotomyAppointment(context.TODO(), "<order_id>", request)

    return err
}
