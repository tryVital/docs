from vital import Client

client = Client(
  api_key=<API_KEY>,
  environment="sandbox",
  region="us"
)

data = client.AtHomePhlebotomy.appointment_availability(
    order_id="413d7205-f8a9-42ed-aa4a-edb99e481ca0",
    address={
        "first_line": "256 West Lincoln Street",
        "second_line": "14",
        "city": "Phoenix",
        "state": "AZ",
        "zip_code": "85004",
    },
)
